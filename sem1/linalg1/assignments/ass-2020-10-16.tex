    \section{Aufgabe (b)}
    
    \begin{align*}
        AB = \begin{pmatrix}
            b_1 - 12 & 2b_2-4 \\
            3b_1 + 6 & -b-12
        \end{pmatrix} = \\
        BA = \begin{pmatrix}
            b_1 - 12 & 2b_1 + 4 \\
            3b_2 - 6 & -b_2-12
        \end{pmatrix}\\
        \Rightarrow \forall b_2 \in \mathbb{R}: b_1 = 4 - b_2 \Rightarrow AB = BA\\ 
        \Leftrightarrow b_1 \neq 4 - b_2 \Rightarrow BA \neq AB
    \end{align*}

    \begin{gather*}
        AC = \begin{pmatrix}
            a + 2c & b + 2d \\
            4a - c & 3b - d
        \end{pmatrix} = \\
        CA = \begin{pmatrix}
            a + 3b & 2a - b \\
            c + 3d & 2c - d
        \end{pmatrix}\\
        \Rightarrow\\ 
        a + 2c = a + 3b, \\
        b + 2d = 2a - b, \\
        4a - c = c + 3d, \\
        3b - d = 2c - d \\
        \Leftrightarrow\\
        3b -2c = 0\\
        a -b -d = 0\\
        4a - 2c - 3d=0\\
        3b - 2c = 0\\
        \Leftrightarrow\\
        \forall \alpha \in \mathbb{R}: a = 0, b = \alpha, c = \frac{3\alpha}2, d = -\alpha \Leftrightarrow AC = CA
    \end{gather*}
